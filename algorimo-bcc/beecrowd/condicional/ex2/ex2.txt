    declare a, b, c, delta, r1, r2 float
    leia a, b, c
    delta = pow(b, 2) -(4 * a * c);
    se (a == 0) ou (delta < 0) entao
        escreva Impossivel calcular
    senao
        r1 = (-b + sqrt(delta)) / (2 * a)
        r2 = (-b - sqrt(delta)) / (2 * a)
        escreva R1 = %.5f\nR2 = %.5f\n , r1, r2
    fim se
fim
